// --- App state ---
let state = {page:0, perPage:10, selections:{}};
const root = document.getElementById('quizRoot');

function render(){
  const start = state.page * state.perPage;
  const end = Math.min(start + state.perPage, QUESTIONS.length);
  root.innerHTML = '';

  const container = document.createElement('div');
  container.className = 'questions';

  for(let i=start;i<end;i++){
    const item = QUESTIONS[i];
    const qwrap = document.createElement('div'); qwrap.className='q';
    const qtitle = document.createElement('div'); qtitle.innerHTML = `<strong>Q${i+1}.</strong> ${item.q}`;
    qwrap.appendChild(qtitle);

    const opts = document.createElement('div'); opts.className='opts';
    item.options.forEach((opt, idx)=>{
      const b = document.createElement('div'); b.className='opt'; b.tabIndex=0; b.innerText = opt;
      // selection highlight
      const key = String(i);
      if(state.selections[key]!==undefined){
        const sel = state.selections[key];
        if(sel === idx){ b.classList.add('selected'); }
        // show correct/wrong styling
        if(sel !== item.correct){
          if(idx === item.correct) b.classList.add('correct');
          if(idx === sel) b.classList.add('wrong');
        } else {
          if(idx === item.correct) b.classList.add('correct');
        }
      }

      b.addEventListener('click', ()=>{ handleSelect(i, idx); });
      opts.appendChild(b);
    });

    qwrap.appendChild(opts);
    container.appendChild(qwrap);
  }

  root.appendChild(container);

  // controls
  const controls = document.createElement('div'); controls.className='controls';
  const pager = document.createElement('div'); pager.className='pager';
  const prev = document.createElement('button'); prev.innerText='Previous'; prev.disabled = state.page===0;
  prev.onclick = ()=>{ state.page = Math.max(0, state.page-1); render(); };
  const next = document.createElement('button'); next.innerText='Next'; next.disabled = end>=QUESTIONS.length;
  next.onclick = ()=>{ state.page++; render(); };
  const pageNum = document.createElement('div'); pageNum.className='page-num'; pageNum.innerText = `Page ${state.page+1} of ${Math.ceil(QUESTIONS.length/state.perPage)}`;
  pager.appendChild(prev); pager.appendChild(pageNum); pager.appendChild(next);

  const right = document.createElement('div');
  const submit = document.createElement('button'); submit.innerText='Submit Page';
  submit.onclick = showPageScore;
  right.appendChild(submit);

  controls.appendChild(pager);
  controls.appendChild(right);
  root.appendChild(controls);

  // overall status
  const answered = Object.keys(state.selections).length;
  const status = document.createElement('div'); status.style.marginTop='10px'; status.innerHTML = `<span class="score">Answered:</span> ${answered} / ${QUESTIONS.length}`;
  root.appendChild(status);
}

function handleSelect(qIndex, optIndex){
  // allow one selection per question; on select, show correct answer (styling applied)
  state.selections[String(qIndex)] = optIndex;
  // re-render to apply correct/wrong styles
  render();
  // after choosing, if wrong, show a small toast with correct
  const correctIdx = QUESTIONS[qIndex].correct;
  if(optIndex !== correctIdx){
    showInlineMessage(qIndex, `Wrong — correct answer: ${QUESTIONS[qIndex].options[correctIdx]}`);
  } else {
    showInlineMessage(qIndex, `Correct!`);
  }
}

function showInlineMessage(qIndex, text){
  // small ephemeral message under that question
  const qEls = document.querySelectorAll('.q');
  const rel = qEls[qIndex - state.page*state.perPage];
  if(!rel) return;
  let msg = rel.querySelector('.inline-msg');
  if(!msg){ msg = document.createElement('div'); msg.className='inline-msg'; msg.style.marginTop='8px'; msg.style.fontSize='13px'; rel.appendChild(msg);} 
  msg.innerText = text;
  setTimeout(()=>{ if(msg) msg.style.opacity = '0.6'; }, 2000);
}

function showPageScore(){
  const start = state.page * state.perPage; const end = Math.min(start + state.perPage, QUESTIONS.length);
  let correct=0, total= end-start;
  for(let i=start;i<end;i++){
    const sel = state.selections[String(i)];
    if(sel !== undefined && sel === QUESTIONS[i].correct) correct++;
  }
  alert(`This page score: ${correct} / ${total}\n(You can navigate pages to continue)`);
}

// initial render
render();
